	bool hasCycle(ListNode *head) {
	    // write your code here
	    if (head == NULL || head -> next == NULL) {
	        return false;
	    }
	    ListNode* p_slow = head;
	    ListNode* p_fast = head;
	    while (p_fast != NULL && p_fast -> next != NULL) {
	        p_slow = p_slow -> next;
	        p_fast = p_fast -> next -> next;
	        if (p_slow == p_fast) {
	            return true;
	        }
	    }
	    return false;
	}

	    fast=head->next;
