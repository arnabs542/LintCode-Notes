	int sqrt(int x) {
	    // write your code here
	    long start = 0;
	    long end = x;
	    long mid;
	    if (x == 1) {
	        return 1;
	    }
	    while (start + 1 < end) {
	        mid = start + (end - start) / 2;
	        if (mid * mid > x) {
	            end = mid;
	        } else if (mid * mid < x) {
	            start = mid;
	        } else {
	            return mid;
	        }
	    }
	    return start;
	}
