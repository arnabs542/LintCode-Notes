	ListNode *detectCycle(ListNode *head) {
	    // write your code here
	    if (head == NULL) {
	        return head;
	    }
	    ListNode* slow = head;
	    ListNode* fast = head;
	    while (fast -> next != NULL && fast -> next -> next != NULL) {
	        slow = slow -> next;
	        fast = fast -> next -> next;
	        if (fast == slow) {
	            slow = head;
	            while (fast != slow) {
	                fast = fast -> next;
	                slow = slow -> next;
	            }
	            return slow;
	        }
	    }
	    return NULL;
	}
